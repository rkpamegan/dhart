126 inline bool CheckGeometryID(HIT_FLAG goal, int id, const GeometryFlagMap & geom_dict) {
127
128 // If the target is both or the geometry rules are set to NO_FLAG, all hits are counted as
129 // being on walkable geometry
130 if (geom_dict.Mode == GeometryFilterMode::ALL_INTERSECTIONS || goal == HIT_FLAG::BOTH)
131 return true;
132
133 // Otherwise do different checks based on the ruleset
134 else if (geom_dict.Mode == GeometryFilterMode::OBSTACLES_ONLY)
135 // If only obstacles are specified, this works like a blacklist/whitelist
136 if (goal == OBSTACLES)
137 return geom_dict[id] == HIT_FLAG::OBSTACLES;
138 else
139 return geom_dict[id] != HIT_FLAG::OBSTACLES;
140
141 else if (geom_dict.Mode == GeometryFilterMode::OBSTACLES_AND_FLOORS)
142 // In OBSTACLES_AND_FLOORS mode, the id's type must exactly match the goal
143 return (goal == geom_dict[id]);
144 }

146 optional_real3 CheckRay(
147 RayTracer& ray_tracer,
148 const real3& origin,
149 const real3& direction,
150 real_t node_z_tolerance,
151 HIT_FLAG flag,
152 const GeometryFlagMap & geometry_dict)
153 {
154 // Setup default params
155 HitStruct<real_t> res;
156
157 // Cast the ray. On success, this returns the ID and distance to intersection.
158 res = ray_tracer.Intersect(origin, direction);
159
160 // Check if it hit and the ID of the geometry matches what we were looking for.
161 if (res.DidHit() && CheckGeometryID(flag, res.meshid, geometry_dict)) {
162 // Create a new optional point with a copy of the origin
163 optional_real3 return_pt(origin);
164
165 // Move it in direction
166 MoveNode(res.distance, direction, *return_pt);
167 //double temp_diff = (origin[2]) - dist; // Sanity check for direction -1 to see influence of movenode arithmetic
168 // Round the position to the z value tolerance
169 return_pt.pt[2] = HF::SpatialStructures::roundhf_tail<real_t>(return_pt.pt[2], 1/node_z_tolerance);
170
171 // Return the optional point
172 return return_pt;
173 }
174
175 // Otherwise, return an empty optional_real3 to signal that
176 // the point didn't intersect
177 return optional_real3();
178 }

180 std::set<std::pair<int, int>> permutations(int limit) {
181 // Create a vector of all numbers between 1 and limit + 1, as well as their inverses
182 vector<int> steps;
183 steps.reserve(2 * limit);
184
185 for (int i = 1; i < limit + 1; i++) {
186 steps.push_back(i);
187 steps.push_back(-i);
188 }
189
190 std::set<std::pair<int, int>> perms;
191
192 // Nest a for loop to capture every permutation
193 for (int j : steps)
194 for (int k : steps)
195 if (abs(j) != abs(k))
196 perms.emplace(pair(j, k));
197
198 return perms;
199 }

290 bool CheckSlope(const real3& parent, const real3& child, const GraphParams& gp)
291 {
292 // Slope is rise/run
293 const real_t run = sqrt(pow((parent[0] - child[0]), 2) + pow((parent[1] - child[1]), 2));
294 const real_t rise = child[2] - parent[2];
295
296 // Calculate the angle between rise and run.
297 const real_t calc_slope = atan2(rise, run) * (static_cast<real_t>(180.00) / static_cast<real_t>(M_PI));
298
299 // Check against downslope and upslope.
300 return calc_slope > -1.0 * gp.down_slope && calc_slope < gp.up_slope;
301 }

385 HF::SpatialStructures::STEP CheckConnection(
386 const real3& parent,
387 const real3& child,
388 RayTracer& rt)
389 {
390 // Default graphh generator ground offset
391 const auto GROUND_OFFSET = 0.01;
392
393 // Create a copy of parent and child that can be modified
394 auto node1 = parent;
395 auto node2 = child;
396
397 // Offset them from the ground slightly
398 node1[2] += GROUND_OFFSET;
399 node2[2] += GROUND_OFFSET;
400
401 if (!OcclusionCheck(node1, node2, rt)) {
402 // Case for when there is direct line of sight between nodes.
403 // Since the edge is already in the graph, we do not need to check whether
404 // they're on the same plane or if the slope between them is valid.
405 return STEP::NONE;
406 }
407 // Otherwise check for step based connections.
408 else {
409 // Since these edges were already verified in the graph generator,
410 // we do not need to check line of sight changes using offsets.
411 STEP s = STEP::NONE;
412
413 // If parent is higher than child, then we have a down step
414 if (parent[2] > child[2]) {
415 s = STEP::DOWN;
416 }
417 // If parent is lower than child, then we have an up step.
418 else if (parent[2] < child[2]) {
419 s = STEP::UP;
420 }
421 // If parent is on an equal plane with the child, then we step over an obstacle.
422 else if (parent[2] == child[2]) {
423 s = STEP::OVER;
424 }
425 return s;
426 }

430 HF::SpatialStructures::STEP CheckConnection(
431 const real3& parent,
432 const real3& child,
433 RayTracer& rt,
434 const GraphParams& params)
435 {
436 // Get groundoffset from graph parameters
437 const auto GROUND_OFFSET = params.precision.ground_offset;
438
439 // Create a copy of parent and child that can be modified
440 auto node1 = parent;
441 auto node2 = child;
442
443 // Offset them from the ground slightly
444 node1[2] += GROUND_OFFSET;
445 node2[2] += GROUND_OFFSET;
446
447 // See if there's a direct line of sight between parent and child
448 if (!OcclusionCheck(node1, node2, rt)) {
449 // If there is a direct line of sight, and they're on the same plane
450 // then there is no step.
451 if (abs(node1[2] - node2[2]) < GROUND_OFFSET) return STEP::NONE;
452
453 // If they are not on the same plane, it means this is a slope. Check if the slope is within the threshold.
454 else if (CheckSlope(parent, child, params)) return STEP::NONE;
455
456 return STEP::NOT_CONNECTED;
457 }
458
459 // Otherwise check for a step based connectiom
460 else {
461 STEP s = STEP::NONE;
462 // If parent is higher than child, the check is to go downstairs
463 // Since the child is lower, raise the child height by the downstep limit
464 // to be checked for a connection
465 if (parent[2] > child[2])
466 {
467 node1 = child;
468 node2 = parent;
469 node1[2] = node1[2] + params.down_step;
470 node2[2] = node2[2] + GROUND_OFFSET;
471 s = STEP::DOWN;
472 }
473
474 // If parent is lower than child, the check is to go upstairs
475 // Since the child is lower, raise the child height by the upstep limit
476 // to be checked for a connection
477 else if (node1[2] < node2[2])
478 {
479 node1 = parent;
480 node2 = child;
481 node1[2] = node1[2] + params.up_step;
482 node2[2] = node2[2] + GROUND_OFFSET;
483 s = STEP::UP;
484 }
485
486 // If they're on an equal plane then offset by upstep to see
487 // if the obstacle can be stepped over.
488 else if (node1[2] == node2[2])
489 {
490 node1 = parent;
491 node2 = child;
492 node1[2] = node1[2] + params.up_step;
493 node2[2] = node2[2] + GROUND_OFFSET;
494 s = STEP::OVER;
495 }
496
497 // If there is a line of sight then the nodes are connected
498 // with the step type we calculated
499 if (!OcclusionCheck(node1, node2, rt))
500 return s;
501 }
502
503 // If not, then there is no connection between these
504 // nodes.
505 return STEP::NOT_CONNECTED;
506 }

509 const GraphParams& params, std::vector < HF::SpatialStructures::EdgeSet> to_compare)
510 {
511 const auto graph_nodes = g.Nodes();
512 for (Node node : graph_nodes) {
513 int parent_id = node.id;

515 for (HF::SpatialStructures::IntEdge edge : outgoing_edges) {
516 int child_id = edge.child;
517 float step_type = edge.weight;
518 Node child = g.NodeFromID(child_id);
519
520 //Type Casting
521 real_t parent_x = CastToReal(node.x);
522 real_t parent_y = CastToReal(node.y);
523 real_t parent_z = CastToReal(node.z);

526 real_t child_y = CastToReal(child.y);
527 real_t child_z = CastToReal(child.z);
528

530 std::vector<real_t> child_loc = { child_x, child_y, child_z };

544 std::vector<real3> GeneratePotentialChildren(
545 const real3& parent,
546 const std::vector<pair>& directions,
547 const real3& spacing,
548 const GraphParams& GP)
549 {
550 // Create a vector of out_children to hold results
551 vector<real3> out_children(directions.size());
552
553 // Iterate through the set of all directions, then offset children in each direction
554 for (int i = 0; i < directions.size(); i++) {
555 // Get the direction at this index
556 const auto& dir = directions[i];
557
558 // Extract the x and y directions
559 real_t x_offset = dir.first;
560 real_t y_offset = dir.second;
561
562 // Add the user-defined spacing to the x and y components of the parent.
563 // Then round the result.
564 const real_t x = roundhf_tmp<real_t>(std::fma(x_offset,spacing[0], parent[0]), GP.precision.node_spacing);
565 const real_t y = roundhf_tmp<real_t>(std::fma(y_offset,spacing[1], parent[1]), GP.precision.node_spacing);
566 // Round the z value to a lower precision assuming it helps embree
567 const real_t z = roundhf_tmp<real_t>(parent[2] + spacing[2], GP.precision.node_z);
568
569 // Add these new values as a node in the out_children list
570 out_children[i] = real3{ x, y, z };
571 }
572
573 return out_children;
