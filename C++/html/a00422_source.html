25C_INTERFACE CreatePath(
26 const HF::SpatialStructures::Graph* g,
27 int start,
28 int end,
29 const char * cost_type,
30 int* out_size,
31 HF::SpatialStructures::Path** out_path,
32 HF::SpatialStructures::PathMember** out_data
33) {
34 // Get a boost graph from *g
35 std::unique_ptr<BoostGraph, BoostGraphDeleter> bg;
36
37 try {
38 // If cost_name is not a valid cost type in *g,
39 // HF::Exceptions::NoCost is thrown by Graph::GetEdges
40 bg = CreateBoostGraph(*g, std::string(cost_type));
41 }
42 catch (HF::Exceptions::NoCost) {
43 return HF::Exceptions::HF_STATUS::NO_COST;
44 }
45 catch (...) { // Ensure that we catch other potential issues
46 return HF::Exceptions::GENERIC_ERROR;
47 }
48
49
50 // Allocate a new empty path
51 Path* P = new Path();
52
53 // Generate a path using the boost graph we just created.
54 *P = FindPath(bg.get(), start, end);
55
56 // If P isn't empty, set our output pointer to it
57 if (!P->empty()) {
58 *out_path = P;
59 *out_data = P->GetPMPointer();
60 *out_size = P->size();
61 return HF::Exceptions::HF_STATUS::OK;
62 }
63
64 // Otherwise, free the memory for it and signal that no path
65 // could be found
66 else {
67 delete P;
68 return HF::Exceptions::HF_STATUS::NO_PATH;
69 }
70}
