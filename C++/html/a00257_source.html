169 template <size_t dimension, typename float_precision = float>
170 bool is_perpendicular(float_precision vec_U[], float_precision vec_V[]) {
171 // Mathematically,
172 // two vectors are perpendicular if their dot product is
173 // equal to zero. But since it is a mortal sin to test
174 // floating point numbers for equality (using operator==) --
175 // we can test if the dot product of
176 // vector_a and vector_b is 'close enough' to zero,
177 // by determining if our dot_product calculation
178 // is less than our ROUNDING_PRECISION constant.
179 // (which is 0.0001)
180 return std::abs(dot_product<dimension, float_precision>(vec_U, vec_V))
181 < HF::SpatialStructures::ROUNDING_PRECISION;
182 }
