169 void Graph::AttrToCost(
170 const std::string& node_attribute,
171 const std::string & out_attribute,
172 Direction gen_using)
173 {
174 // Throw if we don't have this attribute
175 if (!this->HasNodeAttribute(node_attribute))
176 throw std::out_of_range("Node Attribute" + node_attribute + " doesn't exist in the graph!");
177
178 // Delete the cost set that already exists at out_attribute unless out_attribute is the default cost set
179 // in which case throw because we'd be clearing the entire graph.
180 if (!this->IsDefaultName(out_attribute)) {
181 if (this->HasCostArray(out_attribute))
182 this->ClearCostArrays(out_attribute);
183 }
184 else
185 throw std::out_of_range("Cost Set" + out_attribute + " is the default cost of the graph and can't be overwritten!");
186
187 // Get the costs for this attribute and convert every value to float. In the case that a
188 // attribute score could not be converted due to not being a numeric value or not being set
189 // in the first place, those values will be set to -1.
190 vector<float> scores;
191 if (this->IsFloatAttribute(node_attribute))
192 scores = this->GetNodeAttributesFloat(node_attribute);
193 else
194 scores = ConvertStringsToFloat(this->GetNodeAttributes(node_attribute));
195
196 // Iterate through all nodes in the graph
197 for (const auto& parent : ordered_nodes) {
198
199 // If this parent has no score for this attribute, don't do anything
200 if (scores[parent.id] == -1) continue;
201
202 // Get the subgraph for this node
203 const auto subgraph = this->GetIntEdges(parent.id);
204 // Iterate through every edge of this node
205 for (const IntEdge& edge : subgraph)
206 {
207 // If this child has no score for this attribute, skip it.
208 if (scores[edge.child] == -1) continue;
209
210 // Calculate the cost of this node based on the input direction
211 float cost = -1;
212 switch (gen_using) {
213 case Direction::INCOMING:
214 // If the direction is incoming, then the only cost we care about is the cost of
215 // the node that is being traversed to, the child.
216 cost= scores[edge.child];
217 break;
218 case Direction::BOTH:
219
220 // If BOTH is specified, then we sum the costs of both the parent node and the childn\
221 // node since we care about the costs of both
222 cost= scores[edge.child] + scores[parent.id];
223 break;
224
225 case Direction::OUTGOING:
226 // If this is out going, then the score is entirely determined by the parent node
227 // since it is the node being traversed from.
228 cost = scores[parent.id];
229 break;
230 }
231
232 // Add it to the graph as an edge for the cost type specified in out_attribute
233 this->addEdge(parent.id, edge.child, cost, out_attribute);
234 }
235 }
236 }

342 int IMPL_ValueArrayIndex(
343 int parent_id,
344 int child_id,
345 const int* outer_index_ptr,
346 const int* inner_index_ptr
347 ) {
348 // Get the bounds for our search
349 const int search_start_index = outer_index_ptr[parent_id];
350 const int search_end_index = outer_index_ptr[parent_id + 1];
351
352 // Get a pointer to the start and end of our search bounds.
353 const int* search_start_ptr = inner_index_ptr + search_start_index;
354 const int* search_end_ptr = inner_index_ptr + search_end_index;
355
356 // Use an iterator to find a pointer to the value in memory
357 auto itr = std::find(search_start_ptr, search_end_ptr, child_id);
358
359 // Return if we hit the end of our search bounds, indicating that the
360 // edge doesn't exist.
361
362 if (itr == search_end_ptr) {
363 return -1;
364 }
365 // Find the distance between the pointer we found earlier and the
366 // start of the values array
367 else {
368 int index = std::distance(inner_index_ptr, itr);
369
370 return index;
371 }

479 inline vector<Edge> IMPL_UndirectedEdges(const csr& edge_matrix, const int parent_id, const Graph * g) {
480
481 // If N is not in the graph, return an empty array.
482 const int node_id = parent_id;
483 if (node_id < 0) return vector<Edge>();
484
485 // Get the directed edges for this node from calling operator[]
486 vector<Edge> out_edges;
487
488 // Iterate through every other node
489 for (int i = 0; i < edge_matrix.rows(); i++) {
490
491 // Don't look in this node's edge array
492 if (i == node_id) continue;
493
494 // See if there's an edge between I and node_id
495 if (edge_matrix.coeff(i, node_id) != 0) {
496
497 // if so, get its cost
498 float cost = edge_matrix.coeff(i, node_id);
499
500 // Get the node belonging to the child's id
501 Node child_node = g->NodeFromID(i);
502
503 // Construct a new edge and push it back into ot_edges
504 Edge edge{ child_node, cost };
505 out_edges.push_back(edge);
506 }
507 }
508 return out_edges;
509 }

584 inline void Aggregate(float& out_total, float new_value, const COST_AGGREGATE agg_type, int & count)
585 {
586 switch (agg_type) {
587 case COST_AGGREGATE::COUNT:
588
589 // Only increment count if newvalue is equal to zero.
590 // In the case that out_total is zero and count is greater than zero,
591 // multiple new elements have been added, and we must not increment
592 if (new_value > 0 && !(out_total == 0 && count > 0))
593 count++;
594 out_total = count;
595 break;
596 case COST_AGGREGATE::SUM:
597 out_total += new_value;
598 break;
599 case COST_AGGREGATE::AVERAGE: {
600 // If this is zero this function won't work.
601 if (count == 0) count = 1;
602
603 out_total = out_total + (new_value - out_total) / static_cast<float>(count);
604 count++;
605 break;
606 }
607 default:
608 throw std::out_of_range("Unimplemented aggregation type");
609 break;
610 }
611 assert((out_total == 0 || std::isnormal(out_total)));
612 return;
613 }

637 template<typename csr>
638 std::vector<float> Impl_AggregateGraph(COST_AGGREGATE agg_type, int num_nodes, bool directed, const csr & edge_matrix) {
639
640 // Create an array of floats filled with zeros.
641 vector<float> out_costs(num_nodes, 0);
642
643 // If directed is true, then we only need the values in a node's row to calculate it's score.
644 if (directed)
645 for (int k = 0; k < num_nodes; ++k) {
646
647 // Sum all values in the row for node k
648 const float sum = edge_matrix.row(k).sum();
649
650 // Get the count of non_zeros for node k
651 int count = edge_matrix.row(k).nonZeros();
652
653 // Run aggregate once with the sum and count of the non-zeros in row k
654 Aggregate(out_costs[k], sum, agg_type, count);
655
656 }
657
658 // Based on implementation from Eigen's sparse matrix tutorial: Iterating over the nonzero
659 // coefficents: https://eigen.tuxfamily.org/dox/group__TutorialSparse.html If directed is
660 // false, then every edge needs to be iterated through since edges that go to a node also
661 // count towards its score.
662 else {
663 // We need to keep track of count per node as well for this algorithm
664 vector<int> count(num_nodes, 0);
665
666 // Iterate through every node in the graph
667 for (int k = 0; k < num_nodes; ++k) {
668
669 // Iterate through every edge for the node at column k
670 for (typename csr::InnerIterator it(edge_matrix, k); it; ++it)
671 {
672 // Get values from the iterator for this row/col.
673 float cost = it.value();
674 int child = it.col();
675 int parent = it.row();
676
677 // Aggregate costs for both the parent and the child.
678 Aggregate(out_costs[parent], cost, agg_type, count[parent]);
679 Aggregate(out_costs[child], cost, agg_type, count[child]);
680 }
681 }
682 }
683 return out_costs;
684 }

771 vector<Edge> Graph::GetEdgesForNode(int parent_id, bool undirected, const string & cost_type) const
772 {
773 // Get the row of this node
774 const int row = parent_id;
775 const bool default_name = this->IsDefaultName(cost_type);
776
777 std::vector<Edge> outgoing_edges;
778
779 // Get all of the outgoing edges by iterating through the array if it's the default
780 if (default_name) {
781 for (SparseMatrix<float, 1>::InnerIterator it(edge_matrix, row); it; ++it) {
782 const auto col = it.col();
783 float value = it.value();
784
785 outgoing_edges.emplace_back(Edge(NodeFromID(col), value));
786 }
787 }
788 // If using a custom cost, use our own indexing function instead of getting the value from the iterator
789 else {
790 const auto & cost_set = this->GetCostArray(cost_type);
791 std::vector<Edge> out_edges;
792 for (SparseMatrix<float, 1>::InnerIterator it(edge_matrix, row); it; ++it) {
793 const auto col = it.col();
794 auto value = this->GetCostForSet(cost_set, row, col);
795
796 outgoing_edges.emplace_back(Edge(NodeFromID(col), value));
797 }
798 }
799
800 // If this is undirected, we'll need to get a list of incoming edges as well.
801 if (undirected) {
802 vector<Edge> incoming_edges;
803
804 // Default name uses the default edge_matrix member
805 if (default_name)
806 incoming_edges = IMPL_UndirectedEdges(this->edge_matrix, parent_id, this);
807
808 // If we're using a custom type, map a CSR to it and use that instead
809 else
810 incoming_edges = IMPL_UndirectedEdges(this->MapCostMatrix(cost_type), parent_id, this);
811
812 // Combine the incoming and outgoing edges
813 vector<Edge> incoming_and_outgoing(incoming_edges.size() + outgoing_edges.size());

1263
